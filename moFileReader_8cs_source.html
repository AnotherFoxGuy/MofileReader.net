265  private ErrorCode ReadStream(Stream stream)
266  {
267  // Creating a file-description.
268  MoFileInfo moInfo;
269 
270  _lookup = new moLookupList();
271  _lookupContext = new moContextLookupList();
272 
273  // Reference to the List inside moInfo.
274  moInfo.m_translationPairInformation = new moTranslationPairList();
275  var transPairInfo = moInfo.m_translationPairInformation;
276 
277  // Read in all the 4 bytes of fire-magic, offsets and stuff...
278  moInfo.m_magicNumber = stream.ReadUInt(4);
279  moInfo.m_fileVersion = stream.ReadInt(4);
280  moInfo.m_numStrings = stream.ReadInt(4);
281  moInfo.m_offsetOriginal = stream.ReadInt(4);
282  moInfo.m_offsetTranslation = stream.ReadInt(4);
283  moInfo.m_sizeHashtable = stream.ReadInt(4);
284  moInfo.m_offsetHashtable = stream.ReadInt(4);
285 
286  if (moInfo.m_magicNumber == 0 || moInfo.m_numStrings == 0)
287  {
288  _error = "Stream bad during reading. The .mo-file seems to be invalid or has bad descriptions!";
289  return ErrorCode.FileInvalid;
290  }
291 
292  // Checking the Magic Number
293  if (MagicNumber != moInfo.m_magicNumber)
294  {
295  if (MagicReversed != moInfo.m_magicNumber)
296  {
297  _error = "The Magic Number does not match in all cases!";
298  return ErrorCode.MagicnumberNomatch;
299  }
300 
301  moInfo.m_reversed = true;
302  _error = "Magic Number is reversed. We do not support this yet!";
303  return ErrorCode.MagicnumberReversed;
304  }
305 
306  // Now we search all Length & Offsets of the original strings
307  for (var i = 0; i < moInfo.m_numStrings; i++)
308  try
309  {
310  var str = new MoTranslationPairInformation
311  {
312  m_orLength = stream.ReadInt(4),
313  m_orOffset = stream.ReadInt(4)
314  };
315  transPairInfo.Add(str);
316  }
317  catch
318  {
319  _error = "Stream bad during reading. The .mo-file seems to be invalid or has bad descriptions!";
320  return ErrorCode.FileInvalid;
321  }
322 
323  // Get all Lengths & Offsets of the translated strings
324  // Be aware: The Descriptors already exist in our list, so we just mod. refs from the deque.
325  for (var i = 0; i < moInfo.m_numStrings; i++)
326  try
327  {
328  var str = transPairInfo[i];
329  str.m_trLength = stream.ReadInt(4);
330  str.m_trOffset = stream.ReadInt(4);
331  transPairInfo[i] = str;
332  }
333  catch
334  {
335  _error = "Stream bad during reading. The .mo-file seems to be invalid or has bad descriptions!";
336  return ErrorCode.FileInvalid;
337  }
338 
339  // Normally you would read the hash-table here, but we don't use it. :)
340 
341  // Now to the interesting part, we read the strings-pairs now
342  for (var i = 0; i < moInfo.m_numStrings; i++)
343  {
344  // We need a length of +1 to catch the trailing \0.
345  var orLength = transPairInfo[i].m_orLength;
346  var trLength = transPairInfo[i].m_trLength;
347 
348  var orOffset = transPairInfo[i].m_orOffset;
349  var trOffset = transPairInfo[i].m_trOffset;
350 
351  string original, translation;
352 
353  // Original
354  try
355  {
356  stream.Seek(orOffset, SeekOrigin.Begin);
357  original = stream.ReadString(orLength);
358  }
359  catch
360  {
361  _error = "Stream bad during reading. The .mo-file seems to be invalid or has bad descriptions!";
362  return ErrorCode.FileInvalid;
363  }
364 
365  // Translation
366  try
367  {
368  stream.Seek(trOffset, SeekOrigin.Begin);
369  translation = stream.ReadString(trLength);
370  }
371  catch
372  {
373  _error = "Stream bad during reading. The .mo-file seems to be invalid or has bad descriptions!";
374  return ErrorCode.FileInvalid;
375  }
376 
377  var ctxSeparator = original.IndexOf(ContextSeparator);
378 
379  // Store it in the map.
380  if (ctxSeparator == -1)
381  {
382  _lookup[original] = translation;
383  _numStrings++;
384  }
385  else
386  {
387  var l = original.Split(ContextSeparator);
388  if (l.Length != 2)
389  {
390  _error = "Failed to split string";
391  return ErrorCode.Error;
392  }
393 
394  var context = l[0];
395  var id = l[1];
396 
397  if (!_lookupContext.ContainsKey(context))
398  _lookupContext.Add(context, new Dictionary<string, string>());
399 
400  _lookupContext[context][id] = translation;
401  _numStrings++;
402  }
403  }
404 
405  // Done :)
406  return ErrorCode.Success;
407  }

585  private long SwapBytes(long inp)
586  {
587  var b0 = (inp >> 0) & 0xff;
588  var b1 = (inp >> 8) & 0xff;
589  var b2 = (inp >> 16) & 0xff;
590  var b3 = (inp >> 24) & 0xff;
591 
592  return (b0 << 24) | (b1 << 16) | (b2 << 8) | b3;
593  }
